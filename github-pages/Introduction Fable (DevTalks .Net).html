Exemples

Fonctions

let square x = x * x
let sq = square 42

⬇

export function square(x) {
    return x * x | 0;
}
export const sq = square(42);

Currying

let add2times3 = (+) 2 >> (*) 3
let result = add2times3 5

⬇

export const add2times3 = $var1 => 3 * (2 + $var1);
export const result = add2times3(5);

Return statement

let x =
  1
  2

⬇

export const x = (() => {
  1;
  return 2;
})();

Pattern matching

let booleanProxy x =
  match x with
  | true -> true
  | false -> false

booleanProxy true
booleanProxy false

⬇

export function booleanProxy(x) {
  if (x) {
    return true;
  } else {
    return false;
  }
}
booleanProxy(true);
booleanProxy(false);

Requirements

  • dotnet SDK 2.0 or higher

  • An F# editor like like VS Code or Atom with Ionide, Visual Studio for Windows or macOS, Emacs with fsharp-mode or Rider

Sources

MODEL

type Model = int

type Msg =
| Increment
| Decrement

let init() : Model = 0

UPDATE

let update (msg:Msg) (model:Model) =
    match msg with
    | Increment -> model + 1
    | Decrement -> model - 1

VIEW

let view model dispatch =

  R.div []
      [ R.button [ OnClick (fun _ -> dispatch Decrement) ] [ R.str "-" ]
        R.div [] [ R.str (sprintf "%A" model) ]
        R.button [ OnClick (fun _ -> dispatch Increment) ] [ R.str "+" ] ]

PROGRAM

Program.mkProgram init update view
#if DEBUG
|> Program.withDebugger
|> Program.withHMR
#endif
|> Program.withReact "elmish-app"
|> Program.run
